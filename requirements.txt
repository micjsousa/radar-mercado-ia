import os
import requests
import feedparser

TOKEN = os.environ["TELEGRAM_TOKEN"]
CHAT_ID = os.environ["CHAT_ID"]

KEYWORDS = [
    "fed",
    "fomc",
    "powell",
    "cpi",
    "inflation",
    "payroll",
    "nfp",
    "interest rate",
    "treasury",
    "yield",
    "oil",
    "brent",
    "opec",
    "china",
    "selic",
    "ibovespa",
    "dollar"
]

rss = feedparser.parse(
    "https://feeds.finance.yahoo.com/rss/2.0/headline"
)

for noticia in rss.entries[:10]:

    titulo = noticia.title
    titulo_lower = titulo.lower()

    if not any(p in titulo_lower for p in KEYWORDS):
        continue

    impacto_win = "Neutro"
    impacto_wdo = "Neutro"
    relevancia = "MÉDIA"

    if any(x in titulo_lower for x in ["fed", "powell", "fomc"]):
        impacto_win = "ALTO"
        impacto_wdo = "ALTO"
        relevancia = "ALTA"

    if any(x in titulo_lower for x in ["cpi", "inflation", "payroll", "nfp"]):
        impacto_win = "MUITO ALTO"
        impacto_wdo = "MUITO ALTO"
        relevancia = "MUITO ALTA"

    mensagem = f"""
📢 RADAR MERCADO IA

📰 {titulo}

🎯 Relevância: {relevancia}

📈 WIN: {impacto_win}
💵 WDO: {impacto_wdo}

🔗 {noticia.link}
"""

    requests.post(
        f"https://api.telegram.org/bot{TOKEN}/sendMessage",
        data={
            "chat_id": CHAT_ID,
            "text": mensagem
        }
    )

    break
